# Route Handlers

Route Handlers let you create API endpoints using the Web `Request` and `Response` APIs. They replace the old `pages/api/` pattern.

## Convention

Create a `route.ts` file inside the `app/` directory:

<Files>
  <Folder name="app" defaultOpen>
    <Folder name="api" defaultOpen>
      <Folder name="hello" defaultOpen>
        <File name="route.ts" />
      </Folder>
      <Folder name="users" defaultOpen>
        <File name="route.ts" />
        <Folder name="[id]" defaultOpen>
          <File name="route.ts" />
        </Folder>
      </Folder>
    </Folder>
  </Folder>
</Files>

<Callout type="warn">
  A `route.ts` and `page.tsx` cannot coexist in the same directory. Route Handlers handle API requests; pages handle UI.
</Callout>

## HTTP methods

Export functions named after HTTP methods:

```ts title="app/api/users/route.ts"
import { NextResponse } from 'next/server'

// GET /api/users
export async function GET() {
  const users = await db.user.findMany()
  return NextResponse.json(users)
}

// POST /api/users
export async function POST(request: Request) {
  const body = await request.json()

  const user = await db.user.create({
    data: { name: body.name, email: body.email },
  })

  return NextResponse.json(user, { status: 201 })
}
```

## Dynamic routes

Use the same dynamic segment conventions as pages:

```ts title="app/api/users/[id]/route.ts"
import { NextResponse } from 'next/server'

interface Context {
  params: Promise<{ id: string }>
}

export async function GET(request: Request, context: Context) {
  const { id } = await context.params

  const user = await db.user.findUnique({ where: { id } })

  if (!user) {
    return NextResponse.json({ error: 'Not found' }, { status: 404 })
  }

  return NextResponse.json(user)
}

export async function DELETE(request: Request, context: Context) {
  const { id } = await context.params

  await db.user.delete({ where: { id } })

  return new Response(null, { status: 204 })
}
```

      const encoder = new TextEncoder()

    import { cookies } from 'next/headers'

    export async function POST(request: Request) {
      const body = await request.json()
      const cookieStore = await cookies()

      // Set a cookie
      cookieStore.set('session', body.token, {
        httpOnly: true,
        secure: true,
        sameSite: 'lax',
        maxAge: 60 * 60 * 24 * 7, // 1 week
      })

      return Response.json({ success: true })

      const cookieStore = await cookies()

## Caching behavior

| Method | Cached by default? |
|---|---|
| `GET` | ✅ Yes (unless using `cookies()`, `headers()`, or `Request` object) |
| All others | ❌ No |

To opt out of caching for a GET handler:

```ts
export const dynamic = 'force-dynamic'
```
